#include<stdio.h>
#include<stdlib.h>

int main()
{
	const double PI = 3.1415926;
	const double R = 1.5;
	const int H = 3;
	double c, s, v;
	c = 2 * PI*R;
	s = PI * R*R;
	v = s * H;
    
    printf("底面周长：%.2f,底面面积：%.2f,体积：%.2f\n",c,s,v);
    system("pause");
	return 0;
}
