int Fei(int n)
{
	if (n > 2)
		return Fei(n - 1) + Fei(n - 2);
	else if (n == 2)
		return 1;
	else return 0;
}

int Factorial(int n)
{
	if (n > 1)
		return n * Factorial(n - 1);
	else if (n == 1)
		return 1;
	else return 0;
}
