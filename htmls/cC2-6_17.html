void A6()
{
	double i,prize;
	printf("输入当月利润：\n");
	scanf("%lf", &i);
	if (i < 10) prize = i * 0.1;
	else if (i < 20)prize = (i - 10)*0.075 + 1;
	else if (i < 40)prize = (i - 20)*0.05 + 1.75;
	else if (i < 60)prize = (i - 40)*0.03 + 2.75;
	else if (i<100)prize = (i - 60)*0.015 + 3.35;
	else prize = (i - 100)*0.01 + 3.75;
	printf("应发奖金%.4f万元\n", prize);
	return;

}

void A9()
{
	char a[64];

	int i, j;
	

	printf("输入一行字符（\"hjo ig#$f *87';@#$4~^\"）：\n");

	for (i = 0; i < 64; i++) {
		if ((a[i] != '\0')) {
			if ((a[i] >= 64 && a[i] <= 90) || (a[i] >= 96 && a[i] <= 122)) e++;
			else if (a[i] == ' ')s++;
			else if (a[i] >= 48 && a[i] <= 57)n++;

		else break;
	}
	printf("输入了%d个字母，%d个数字，%d个空格，其他字符%d个。\n",e,n,s,o);

}

void A11()
{
	double e = 1,e_1;
	int n=1;
	while (1){
		e_1 = 1./Factorial(n);
		if (e_1 > 10e-5)e += e_1;
		else break;
		n++;
	}
	printf("e的值：%f", e);
}

int Factorial(int n)
{
	if (n == 1)return 1;
	else return n * Factorial(n - 1);
}


void A12()
{
	int a,n,i;
	int sum = 0;
	printf("输入a与n(a n):\n");
	scanf("%d %d",&a,&n);
	for (i = 1; i <= n; i++)
		sum += NAA(i, a);
	printf("和为%d\n", sum);
	return;
}

int NAA(int n, int a) {
	int i;
	int sum=0;
	for (i = 0; i < n; i++) {
		sum += (int)pow(10,i)*a;
	}
	return sum;
}
